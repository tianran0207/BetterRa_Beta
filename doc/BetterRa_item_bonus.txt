//= BetterRa 是基于 rAthena 進行二次開發的, 在 rAthena 官方支
//= 持的物品附加效果基礎上, BetterRa 也拓展了一些物品附加效果, 請參閱
//= 下面的說明, 它會介紹這些物品附加效果的使用方法.

bonus4 bSkillForAOE,sk,range,n,r;		使用sk技能時, 有r(千分率)的几率對range范圍內的n個目標使用同樣的技能.
bonus bWeaponAddDamage,n;				普通攻擊或武器技能攻擊時增加 n 的固定傷害
bonus bMagicAddDamage,n;				魔法技能攻擊時增加 n 的固定傷害
bonus bMiscAddDamage,n;					混合技能攻擊時增加 n 的固定傷害
bonus bAddMastery,n; 					增加掌控傷害 n, 例斧修煉額外增加等級 * 5傷害 bonus bAddMastery,5*getskilllv(NC_TRAININGAXE);
bonus3 bSkillDamage,sk,lv,n;			為lv等級的sk技能增加n的固定傷害.
											等級為-1時 每無論多少級都會增加固定傷害值
											等級為 0時 按照技能等級倍數增加固定傷害
											等級為>0時 只對該技能等級增加傷害
	
bonus2 bSkillTreeCooldownRate,jobid,n;	職業技能公共技能延遲按百分比調整
bonus2 bSkillTreeCooldown,jobid,n;		職業技能公共技能延遲調整
bonus2 bSkillTreeDelayRate,jobid,n;		職業技能獨立技能延遲按百分比調整
bonus2 bSkillTreeDelay,jobid,n;			職業技能獨立技能延遲調整

bonus2 bReborn,r,n;						當玩家死亡時有 r% 的概率复活并恢复 n% 的血量
bonus2 bBlockStatus,x,n;				固定玩家屬性 x 為 n
										x 支持	STATUS_MAXHP STATUS_MAXSP 
												STATUS_STR STATUS_AGI STATUS_VIT STATUS_INT STATUS_DEX STATUS_LUK
												STATUS_ATK_L STATUS_ATK_R STATUS_MATK_L STATUS_MATK_R
												STATUS_HIT STATUS_SPEED STATUS_FLEE_L STATUS_FLEE_R STATUS_CRI
												STATUS_DEF_L STATUS_DEF_R STATUS_MDEF_L STATUS_MDEF_R
												STATUS_ATK_ELE STATUS_DEF_ELE STATUS_ATK_RANGE STATUS_ASPD

bonus bCooldownrate,n;					技能獨立延遲時間 + n%
bonus bMaxASPD,n;						修改ASPD上限(無視CONF中的配置)
bonus bDefSize,s;						修改体型大小為 s
bonus bDefRace,r;						修改种族為 r
bonus bDefEleLevel,n;					修改防御屬性等級為 n
bonus3 bSkillItemFix,sk,iid,n;			技能 sk 消耗的物品ID iid 增加 n 個(n可以為負數)
bonus3 bRegen,hp,sp,t;					每隔 t 毫秒, 恢复 HP 和 SP (不覆蓋, 獨立運算)
bonus3 bRegenPercent,hp,sp,t;			每隔 t 毫秒, 按百分比恢复 HP 和 SP(不覆蓋, 獨立運算)

bonus bFixedCastAddRate,n;				固定詠唱時間 + n%(可累加)
bonus3 bMultiDamageRate,bf,r,n;			當玩家使用 bf 攻擊時, 有 r/10% 的概率使傷害變為原來的 n% 倍
bonus3 bSkillCritical,sk,r,n;			當玩家使用 sk 攻擊時, 有 r/10% 的概率使傷害變為原來的 n% 倍
										sk 為 -1 時所有技能暴擊
bonus bSkillCritAtkRate,n;				技能暴擊傷害 + n%
bonus3 bSkillAddDiv,sk,div,n;			技能 sk 打擊段數增加 div, 傷害 + n%
bonus2 bSkillChangeType,sk,bf;          修改技能 sk 攻擊類型為 bf

bonus2 bIgnoreMdefRaceVal,r,n;			無視 r 种族 n 點魔法防御力
bonus2 bIgnoreMdefClassVal,c,n;			無視 c 類別魔物 n 點魔法防御力
bonus2 bIgnoreDefRaceVal,r,n;			無視 r 种族 n 點魔法防御力
bonus2 bIgnoreDefClassVal,c,n;			無視 c 類別魔物 n 點魔法防御力

bonus bIgnoreWhiteImprison,n;			無視白色監獄, n無意義

bonus3 bFinalAddRace,r,x,bf;			使用 bf 攻擊時, 增加 r 种族 x% 的傷害(在最終傷害上全段修正)
bonus3 bFinalAddClass,c,n,bf;			使用 bf 攻擊時, 增加 c 類別魔物 x% 的傷害(在最終傷害上全段修正)

bonus2 bAddSkillArea,sk,n;				增加 sk 技能 n 格范圍, 可用于濺射或范圍技能
bonus2 bAddSkillRange,sk,n;				增加 sk 技能 n 格攻擊距离

bonus2 bSkillToAround,sk,n;				技能 sk 變為群体技能, 自動選擇自身 n 格范圍內的敵對目標進行攻擊
bonus2 bSkillToGround,sk,n;				技能 sk 變為地面技能, 并攻擊 n 格范圍內的敵對目標

bonus2 bSkillNoSpecial,sk,bitmask;		移除技能 sk 的特殊條件, bitmask 為 掩碼類型(可疊加),  
										轉生術/十字驅魔 (bitmask=1 移除對方為不死或惡魔限制)
										水球術/天使之淚 (bitmask=1 移除地面為水面限制)
										背刺(bitmask=1 移除背對限制)
										阿修羅霸皇拳(bitmask=4 移除SP消耗)
										(其他技能, 可聯系作者添加)
										
bonus2 bSkillNoRequire,sk,bitmask;		移除技能 sk 的需求條件, bitmask 為 掩碼類型(可疊加),  
										移除 'skill_require_db.txt' 中技能的指定需求.
										   1 = hp
										   2 = maxhptrigger
										   4 = sp
										   8 = hprate
										  16 = sprate
										  32 = zeny
										  64 = weapon type
										 128 = ammo (with the amount)
										 256 = state
										 512 = statuses
										1024 = spirit sphere
										2048 = items (with the amount)
										4096 = equipments
										
bonus3 bSkillAttribute,sk,ele,n;		技能 sk 變為 ele 屬性(為避免帶寬消耗過大或特效卡客戶端, 每個技能最多顯示一次溫暖風特效)
										 sk 為  0 時為普通攻擊(n 為是否顯示溫暖的風特效)
										 sk 為 -1 時為所有技能(n 為是否顯示溫暖的風特效)
										ele 為 -1 時變為最克制屬性(n 為是否顯示溫暖的風特效), 
										ele 為 -2 時變為最接近屬性(n 為是否顯示溫暖的風特效), 
										ele 為 -3 時無視屬性克制(傷害倍率修正為 n%, 优先度最高),

bonus bWeaponReturnDefRate,n;			受到物理反射傷害時, 傷害減少原來的 n%
bonus bMagicReturnDefRate,n;			受到魔法反射傷害時, 傷害減少原來的 n%

bonus3 bStatusAddBonus,sc,bonus,n;		當自己擁有 sc 狀態的時候, 會增加 bonus 參數為 n 
										例: bonus3 bStatusAddBonus,SC_BLESSING,bStr,20;
										說明: 天使賜福額外增加 20點 STR
bonus4 bStatusAddBonus,sc,bonus,r,n;	當自己擁有 sc 狀態的時候, 會增加 bonus 參數為 r,n
										例: bonus4 bStatusAddBonus,SC_BLESSING,bSkillHeal,AL_HEAL,100;
										說明: 天使賜福額外增加 治愈術回复量 100%
										
bonus5 bAddStatusVal,sc,val1,val2,val3,val4;	賦予狀態時, 參數分別累加 val1, val2, val3, val4
												例: bonus5 bAddStatusVal,SC_BLESSING,10,10,0,0;
												說明: 賦予天使賜福狀態時 STR DEX INT 分別加 20 點
